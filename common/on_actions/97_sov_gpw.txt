on_actions = {
	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM = { tag = SOV }
					ROOT = { 
						OR = { 
							is_in_faction_with = GER 
							tag = GER
						} 
					}
					FROM.FROM = {
						has_state_flag = SOV_controls
					}
				}
				FROM.FROM = {  
				 	set_state_flag = GER_controls
				 	clr_state_flag = SOV_controls
				}
				if = {
					limit = { FROM.FROM = { has_dynamic_modifier = { modifier = SOV_defend_city } } }
					FROM.FROM = { remove_dynamic_modifier = { modifier = SOV_defend_city } }
				}
		 		SOV = {
		 			SOV_great_patriotic_increase = yes 
		 		}
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM = { 
					    OR = { 
							is_in_faction_with = GER
							tag = GER
						}
					}
					ROOT = { tag = SOV }
					FROM.FROM = {
						has_state_flag = GER_controls
					}
				}
				FROM.FROM = { 
				 	set_state_flag = SOV_controls 
				 	clr_state_flag = GER_controls 
				}
		 		SOV = {
		 			SOV_great_patriotic_decrease = yes 
		 		}
				if = { limit = { state = 195 } add_to_variable = { global.leningrad_fall_count = 1 } }
				if = { limit = { state = 202 } add_to_variable = { global.kiev_fall_count = 1 } }
				if = { limit = { state = 217 } add_to_variable = { global.stalingrad_fall_count = 1 } }
				if = { limit = { state = 219 } add_to_variable = { global.moscow_fall_count = 1 } }
			}
		}
	}
}
