add_namespace = ABDA_events

# ABDA_events.1 - Invitation to join ABDACOM
# ROOT = receiving nation, FROM = nation that completed the focus
country_event = {
	id = ABDA_events.1
	title = ABDA_events.1.t
	desc = ABDA_events.1.desc
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes

	option = {
		name = ABDA_events.1.a
		ai_chance = {
			base = 100
			modifier = {
				add = -20
				NOT = {
					OR = {
						tag = ENG
						tag = USA
						tag = HOL
						tag = INS
					}
				}
			}
		}
		set_country_flag = abdacom_member
		add_dynamic_modifier = { modifier = ABDA_supreme_command }
		hidden_effect = {
			set_variable = { ABDA_repair_speed_factor = 0 }
			set_variable = { ABDA_convoy_escort_efficiency = 0 }
			set_variable = { ABDA_submarine_attack_factor = 0 }
			set_variable = { ABDA_submarine_defence_factor = 0 }
			set_variable = { ABDA_screening_efficiency = 0 }
			set_variable = { ABDA_screen_defence_factor = 0 }
			set_variable = { ABDA_experience_gain_navy_unit_factor = 0 }
		}
		FROM = { country_event = { id = ABDA_events.2 } }
	}

	option = {
		name = ABDA_events.1.b
		ai_chance = {
			base = 0
			modifier = {
				add = 20
				NOT = {
					OR = {
						tag = ENG
						tag = USA
						tag = HOL
						tag = INS
					}
				}
			}
		}
		set_country_flag = abdacom_declined
		FROM = { country_event = { id = ABDA_events.3 } }
	}
}

# ABDA_events.4 - Fortress Singapore fortification choice (fires at ENG)
# ROOT = ENG

	id = ABDA_events.4
	title = ABDA_events.4.t
	desc = ABDA_events.4.desc
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes

	# Naval focus: coastal forts + AA
	option = {
		name = ABDA_events.4.a
		1021 = {
			add_building_construction = {
				type = coastal_bunker
				level = 4
				instant_build = yes
				province = { all_provinces = yes limit_to_naval_base = yes }

			}
		}
